<script src="jquery.dom-outline-1.0.js"></script>

var myDomOutline = false;
var editor = CodeMirror.fromTextArea(html, {
  mode: 'text/html', 
  theme: 'neo',
  lineNumbers: true
});
var render = function(){ 
  if(myDomOutline) myDomOutline.stop();
  $('#render').html(editor.getValue()); 
};
editor.on('changes',render);
render();
